To get the application URL:

{{- if eq .Values.app.service.type "LoadBalancer" }}

  NOTE: It may take a few minutes for the LoadBalancer IP to be available.

  export SERVICE_IP=$(kubectl get svc {{ .Values.app.name }} -n {{ .Release.Namespace }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "Application URL: http://$SERVICE_IP:{{ .Values.app.service.port }}"

{{- else if eq .Values.app.service.type "ClusterIP" }}

  export POD_NAME=$(kubectl get pods -n {{ .Release.Namespace }} -l "app={{ .Values.app.name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl port-forward $POD_NAME -n {{ .Release.Namespace }} 8080:{{ .Values.app.service.targetPort }}

  Then visit: http://localhost:8080

{{- end }}
